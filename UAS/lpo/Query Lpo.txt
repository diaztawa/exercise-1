select tb_pelanggan.nama, tb_pelanggan.no_meter, tb_pelanggan.alamat, tb_penggunaan.bulan, tb_penggunaan.tahun, (tb_penggunaan.meterakhir - tb_penggunaan.meterawal) as jmlmeteran, ((tb_penggunaan.meterakhir - tb_penggunaan.meterawal)*tb_pelanggan.kd_tarif) as totalharga from tb_pelanggan, tb_penggunaan, tb_tagihan where tb_pelanggan.id_pelanggan = tb_penggunaan.id_pelanggan AND tb_pelanggan.id_pelanggan = tb_tagihan.id_pelanggan group by tb_pelanggan.id_pelanggan, tb_penggunaan.bulan
